import React, { useState, useEffect, useRef } from 'react';
import { Button } from 'antd';
import { LinearGradient } from 'expo-linear-gradient';

interface SMSVerifyCodeInputProps {
    onInputCompleted: (code: string) => void;
    onVerificationSend: () => void;
    autoStartCountdown?: boolean;
}

export default function SMSVerifyCodeInput({ 
    onInputCompleted, 
    onVerificationSend,
    autoStartCountdown = false 
}: SMSVerifyCodeInputProps) {
    const [code, setCode] = useState(['', '', '', '']);
    const [activeIndex, setActiveIndex] = useState(0);
    const [countdown, setCountdown] = useState(0);
    const inputRefs = useRef<(HTMLInputElement | null)[]>([]);

    useEffect(() => {
        if (autoStartCountdown) {
            setCountdown(60);
            onVerificationSend();
        }
    }, [autoStartCountdown, onVerificationSend]);

    useEffect(() => {
        if (countdown > 0) {
            const timer = setTimeout(() => setCountdown(countdown - 1), 1000);
            return () => clearTimeout(timer);
        }
    }, [countdown]);

    const handleCodeChange = (value: string, index: number) => {
        const newCode = [...code];
        newCode[index] = value;
        setCode(newCode);

        if (value && index < 3) {
            setActiveIndex(index + 1);
            inputRefs.current[index + 1]?.focus();
        }

        if (newCode.every(c => c)) {
            onInputCompleted(newCode.join(''));
        }
    };

    const handleKeyDown = (e: React.KeyboardEvent, index: number) => {
        if (e.key === 'Backspace' && !code[index] && index > 0) {
            setActiveIndex(index - 1);
            inputRefs.current[index - 1]?.focus();
        }
    };

    const handleCellClick = (index: number) => {
        setActiveIndex(index);
        inputRefs.current[index]?.focus();
    };

    const handleSendVerification = () => {
        if (countdown === 0) {
            setCountdown(60);
            onVerificationSend();
        }
    };

    return (
        <div style={styles.container}>
            <div style={styles.codeContainer}>
                {code.map((digit, index) => (
                    <div
                        key={index}
                        style={[
                            styles.cell,
                            activeIndex === index && styles.activeCell
                        ]}
                        onClick={() => handleCellClick(index)}
                    >
                        <input
                            ref={el => inputRefs.current[index] = el}
                            style={styles.input}
                            type="text"
                            maxLength={1}
                            value={digit}
                            onChange={(e) => handleCodeChange(e.target.value, index)}
                            onKeyDown={(e) => handleKeyDown(e, index)}
                        />
                    </div>
                ))}
            </div>

            <Button
                style={[styles.sendButton, countdown > 0 && styles.sendButtonDisabled]}
                onClick={handleSendVerification}
                disabled={countdown > 0}
            >
                <LinearGradient
                    colors={['#6A4C93', '#20172D']}
                    start={{ x: 0, y: 0 }}
                    end={{ x: 0, y: 1 }}
                    style={styles.gradient}
                >
                    <span style={styles.sendButtonText}>
                        {countdown > 0 ? `Resend in ${countdown}s` : 'Send verification code'}
                    </span>
                </LinearGradient>
            </Button>
        </div>
    );
}

const styles = {
    container: {
        display: 'flex',
        flexDirection: 'column',
        alignItems: 'center',
        width: '100%',
    },
    codeContainer: {
        display: 'flex',
        flexDirection: 'row',
        justifyContent: 'center',
        gap: '12px',
    },
    cell: {
        width: '60px',
        height: '60px',
        backgroundColor: '#ffffff',
        borderRadius: '8px',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        cursor: 'pointer',
    },
    activeCell: {
        border: '2px solid #6A4C93',
    },
    input: {
        width: '100%',
        height: '100%',
        border: 'none',
        outline: 'none',
        textAlign: 'center',
        fontSize: '24px',
        backgroundColor: 'transparent',
    },
    sendButton: {
        height: '60px',
        borderRadius: '8px',
        overflow: 'hidden',
        border: 'none',
        cursor: 'pointer',
        width: '50%',
        marginTop: '12px',
    },
    sendButtonDisabled: {
        opacity: 0.5,
        cursor: 'not-allowed',
    },
    gradient: {
        width: '100%',
        height: '100%',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
    },
    sendButtonText: {
        color: '#fff',
        fontSize: '18px',
        fontWeight: 600,
    },
};
